```python
#!/usr/bin/env python3
"""
source_corrob_min_packet_validator
--------------------------------

Deterministic, local-only validator direction for `source-corroboration-runtime`.

This validator is intentionally tiny: it defines the *boundary* of what a future
source corroboration runtime should consider minimally well-formed before any
heavier orchestration (web search, retrieval, clustering) exists.

Contract (v0 direction):
- claim exists and is a JSON object
- at least one evidence item exists
- each evidence item has a provenance/source ref string
- verification_report contains evidence refs (artifact://... strings)

Non-goals:
- network access (explicitly forbidden)
- truth evaluation of claim content
- citation completeness beyond the minimal provenance ref
"""

from __future__ import annotations

import argparse
import json
from dataclasses import dataclass
from pathlib import Path
from typing import Any


@dataclass(frozen=True)
class ValidationOk:
    ok: bool
    errors: list[str]


def _load_json(path: Path) -> Any:
    return json.loads(path.read_text(encoding="utf-8"))


def validate_min_claim(obj: Any) -> list[str]:
    errors: list[str] = []
    if not isinstance(obj, dict):
        return ["claim must be a JSON object"]
    claim_text = obj.get("claim")
    if not isinstance(claim_text, str) or not claim_text.strip():
        errors.append("claim.claim must be a non-empty string")
    return errors


def validate_min_evidence_packet(obj: Any) -> list[str]:
    errors: list[str] = []
    if not isinstance(obj, dict):
        return ["evidence_packet must be a JSON object"]
    evidence = obj.get("evidence")
    if not isinstance(evidence, list) or len(evidence) < 1:
        errors.append("evidence_packet.evidence must be a non-empty array")
        return errors
    for idx, item in enumerate(evidence):
        if not isinstance(item, dict):
            errors.append(f"evidence_packet.evidence[{idx}] must be an object")
            continue
        provenance = item.get("provenance") or item.get("source_ref") or item.get("source")
        if not isinstance(provenance, str) or not provenance.strip():
            errors.append(f"evidence_packet.evidence[{idx}] must include a provenance/source ref string")
    return errors


def validate_verification_report_includes_evidence_refs(obj: Any) -> list[str]:
    errors: list[str] = []
    if not isinstance(obj, dict):
        return ["verification_report must be a JSON object"]
    refs = obj.get("validated_artifact_refs")
    if not isinstance(refs, list) or len(refs) < 1:
        return ["verification_report.validated_artifact_refs must be a non-empty array"]
    has_evidence_ref = any(isinstance(ref, str) and ref.startswith("artifact://") for ref in refs)
    if not has_evidence_ref:
        errors.append("verification_report.validated_artifact_refs must include at least one artifact:// ref")
    return errors


def validate(
    *,
    claim: Any,
    evidence_packet: Any,
    verification_report: Any,
) -> ValidationOk:
    errors: list[str] = []
    errors.extend(validate_min_claim(claim))
    errors.extend(validate_min_evidence_packet(evidence_packet))
    errors.extend(validate_verification_report_includes_evidence_refs(verification_report))
    return ValidationOk(ok=len(errors) == 0, errors=errors)


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description="Validate minimal source corroboration packet fixtures (local-only).")
    parser.add_argument("--claim", required=True, help="Path to claim fixture JSON")
    parser.add_argument("--evidence-packet", required=True, help="Path to evidence packet fixture JSON")
    parser.add_argument("--verification-report", required=True, help="Path to verification report JSON")
    args = parser.parse_args(argv)

    claim = _load_json(Path(args.claim))
    evidence = _load_json(Path(args.evidence_packet))
    report = _load_json(Path(args.verification_report))

    result = validate(claim=claim, evidence_packet=evidence, verification_report=report)
    if not result.ok:
        print(json.dumps({"ok": False, "errors": result.errors}, indent=2))
        return 1
    print(json.dumps({"ok": True}, indent=2))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

```
