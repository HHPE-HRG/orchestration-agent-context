#!/usr/bin/env python3
"""
Guardrail YAML matrix hygiene (Lane D / **U11**).

Ensures adapters listed in maturity contracts stay present under `docs/policies/
guardrail-failure-modes.yaml` so new adapter tooling cannot silently drop policy rows.

This intentionally checks structure only (`modes.ci` existence), not YAML semantics parity
with executor code paths (that's **U10** convergence work).
"""

from __future__ import annotations

import sys
from pathlib import Path


def main(argv: list[str] | None = None) -> int:
    try:
        import yaml  # type: ignore
    except ImportError:
        print("guardrail-policy-check: PyYAML required (installed via uv sync).", file=sys.stderr)
        return 2

    repo_root = Path(__file__).resolve().parents[1]
    yaml_path = repo_root / "docs" / "policies" / "guardrail-failure-modes.yaml"
    if not yaml_path.exists():
        print(f"missing guardrail YAML: {yaml_path}", file=sys.stderr)
        return 1

    data = yaml.safe_load(yaml_path.read_text(encoding="utf-8"))
    if not isinstance(data, dict):
        print("guardrail YAML root must be a mapping", file=sys.stderr)
        return 1
    adapters = data.get("adapters")
    if not isinstance(adapters, dict):
        print("guardrail YAML missing adapters: mapping", file=sys.stderr)
        return 1

    required = sorted(
        {
            "validate",
            "import",
            "approve",
            "activate",
            "sync",
            "evaluate",
            "replan",
            "replay",
            "aggregate",
            "mark_terminal",
        },
    )
    failures: list[str] = []
    for key in required:
        entry = adapters.get(key)
        if not isinstance(entry, dict):
            failures.append(f"adapter `{key}` missing or not a mapping")
            continue
        modes = entry.get("modes")
        if not isinstance(modes, dict):
            failures.append(f"adapter `{key}` missing modes mapping")
            continue
        if "ci" not in modes:
            failures.append(f"adapter `{key}` missing modes.ci")

    extras = sorted(set(adapters.keys()) - set(required))
    if extras:
        print(
            "\n".join(f"WARN: unrecognized adapter keys not in Tier-1 allowlist: {extras}"),
            file=sys.stderr,
        )

    if failures:
        print("guardrail policy matrix failures:\n" + "\n".join(f"  - {f}" for f in failures), file=sys.stderr)
        return 1
    print("guardrail-policy-check: OK")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

#!/usr/bin/env python3
"""
Emit `orchestration_maturity_certificate.json` (ADR 0007).

Intended for `.github/workflows/orchestration-maturity-certificate.yml`.
`gates.*` default to **skipped** until CI supplies `GATE_*` environment pins
(or callers pass `--aggregate-gates`).
"""

from __future__ import annotations

import argparse
import json
import os
import sys
from datetime import UTC, datetime
from pathlib import Path
from typing import Any


def _env_gate(name: str, default: str = "skipped") -> str:
    return os.environ.get(name, default)


def build_certificate(
    *,
    xlotyl_git_sha: str,
    stoneforge_git_sha: str,
    orchestration_agent_context_git_sha: str,
) -> dict[str, Any]:
    now = datetime.now(tz=UTC).isoformat().replace("+00:00", "Z")

    return {
        "certificate_schema_version": "1.0.0",
        "generated_at": now,
        "xg_repos": {
            "xlotyl_git_sha": xlotyl_git_sha,
            "stoneforge_git_sha": stoneforge_git_sha,
            "orchestration_agent_context_git_sha": orchestration_agent_context_git_sha,
        },
        "schema_pins": [],
        "package_lock_hashes": [],
        "gates": {
            "state_recovery": _env_gate("GATE_STATE_RECOVERY"),
            "director_guard": _env_gate("GATE_DIRECTOR_GUARD"),
            "tracing_breadth": _env_gate("GATE_TRACING_BREADTH", "pass"),
            "adapter_guardrails": _env_gate("GATE_ADAPTER_GUARDRAILS"),
        },
        "evidence_links": json.loads(os.environ.get("CERT_EVIDENCE_LINKS_JSON", "[]")),
    }


def gate_failures(doc: dict[str, Any]) -> list[tuple[str, str]]:
    gates = doc.get("gates") or {}
    return [(key, status) for key, status in gates.items() if status == "fail"]


def print_gate_table(doc: dict[str, Any]) -> None:
    gates = doc.get("gates") or {}
    rows = [["gate_id", "status"], *sorted((k, str(v)) for k, v in gates.items())]
    widths = [max(len(r[i]) for r in rows) for i in range(2)]
    for row in rows:
        line = f"{row[0].ljust(widths[0])}  {row[1].ljust(widths[1])}"
        print(line, file=sys.stderr)


def validate_certificate_schema(doc: dict[str, Any]) -> None:
    """
    Validates against ``schemas/ci/orchestration-maturity-certificate.schema.json``.

    Raises ``RuntimeError`` when ``jsonschema`` is not installed (callers translate to exit 3).
    """

    try:
        import jsonschema  # type: ignore[import-not-found]
    except ImportError as err:  # pragma: no cover - exercised via optional install
        raise RuntimeError("install jsonschema when using --validate-schema") from err

    repo_root = Path(__file__).resolve().parent.parent
    schema_path = repo_root / "schemas/ci/orchestration-maturity-certificate.schema.json"
    schema = json.loads(schema_path.read_text(encoding="utf-8"))
    jsonschema.validate(instance=doc, schema=schema)


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description="Emit orchestration maturity certificate JSON.")
    parser.add_argument(
        "--out",
        default=str(Path("orchestration_maturity_certificate.json")),
        help="Output path (default ./orchestration_maturity_certificate.json)",
    )
    parser.add_argument(
        "--fail-on-gate-fail",
        action="store_true",
        help="Exit 1 when any gates.* equals 'fail'; prints stderr summary table.",
    )
    parser.add_argument(
        "--validate-schema",
        action="store_true",
        help="Validate emitted JSON against schemas/ci/orchestration-maturity-certificate.schema.json (requires jsonschema).",
    )
    args = parser.parse_args(argv)

    xlotyl_sha = os.environ.get("GITHUB_SHA", os.environ.get("XLOTYL_GIT_SHA", "local-dev"))
    sf_sha = os.environ.get("STONEFORGE_GIT_SHA", "unspecified")
    oac_sha = os.environ.get("ORCHESTRATION_AGENT_CONTEXT_GIT_SHA", "unspecified")

    doc = build_certificate(
        xlotyl_git_sha=xlotyl_sha,
        stoneforge_git_sha=sf_sha,
        orchestration_agent_context_git_sha=oac_sha,
    )
    out_path = Path(args.out)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(doc, indent=2) + "\n", encoding="utf-8")

    try:
        if args.validate_schema:
            validate_certificate_schema(doc)
        if args.fail_on_gate_fail and gate_failures(doc):
            print("orchestration_maturity_certificate_emit: gates failed:", file=sys.stderr)
            print_gate_table(doc)
            return 1
    except RuntimeError as err:
        print(str(err), file=sys.stderr)
        return 3
    except Exception as err:
        print(f"certificate emitter aborted ({type(err).__name__}): {err}", file=sys.stderr)
        print_gate_table(doc)
        return 2

    print(f"wrote {args.out}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

/**
 * Copy Stoneforge director-path-register.json into Xlotyl `fixtures/ci/` for contract tests / CI parity (R3 / D1).
 *
 * Sources (first existing wins):
 * 1. `STONEFORGE_DIRECTOR_REGISTER_JSON`
 * 2. `<repo>/../stoneforge/docs/orchestration/director-path-register.json` when run from xlotyl root
 */

import { copyFileSync, mkdirSync, statSync } from "node:fs"
import { dirname, join } from "node:path"
import { fileURLToPath } from "node:url"

const scriptDir = dirname(fileURLToPath(import.meta.url))
const xlotylRoot = join(scriptDir, "..")
const destDir = join(xlotylRoot, "fixtures", "ci")
const dest = join(destDir, "director-path-register.json")

const forced = process.env.STONEFORGE_DIRECTOR_REGISTER_JSON
const defaultCandidate = join(xlotylRoot, "..", "stoneforge", "docs", "orchestration", "director-path-register.json")

const candidates = [...(forced ? [forced] : []), defaultCandidate]

let src = null
for (const p of candidates) {
  try {
    statSync(p)
    src = p
    break
  } catch {
    /* try next */
  }
}

if (!src) {
  console.error("sync-director-register: could not find source JSON. Set STONEFORGE_DIRECTOR_REGISTER_JSON.")
  process.exit(1)
}

mkdirSync(destDir, { recursive: true })
copyFileSync(src, dest)
console.error(`sync-director-register: copied\n  ${src}\n→ ${dest}`)
